    //! key at DNS label `_cipherpost` (crate::DHT_LABEL_OUTER). Inner signature is
    //! Ed25519 over the JCS-canonical serialization of `OuterRecordSignable` (this
    //! module). Outer signature is PKARR's SignedPacket signature (transport layer).
    //!
    //! Pitfalls addressed:
    //!   #3 — canonical JSON via serde_canonical_json (JCS / RFC 8785)
    //!   #11 — share_ref is bound to ciphertext + created_at (no server-side enforcement needed)
    //!
    //! D-16: all signature-verification failures return an Error variant that
    //! Displays as "signature verification failed"; internal variants are distinct
    //! so tests can assert which check fired.

    use crate::error::Error;
    use crate::PROTOCOL_VERSION;
    use base64::Engine;
    use ed25519_dalek::{Signature, Verifier, VerifyingKey};
    use serde::{Deserialize, Serialize};
    use serde_canonical_json::CanonicalFormatter;
    use sha2::{Digest, Sha256};

    pub const SHARE_REF_BYTES: usize = 16;
    pub const SHARE_REF_HEX_LEN: usize = SHARE_REF_BYTES * 2;

    /// Signed form — what goes in a DNS TXT record under label `_cipherpost`.
    /// Fields MUST stay in alphabetical order both in this struct and in OuterRecordSignable.
    #[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Eq)]
    pub struct OuterRecord {
        pub blob: String,
        pub created_at: i64,
        pub protocol_version: u16,
        pub pubkey: String,
        pub recipient: Option<String>,
        pub share_ref: String,
        pub signature: String,
        pub ttl_seconds: u64,
    }

    /// Unsigned form — the exact bytes Ed25519 signs are `jcs(OuterRecordSignable)`.
    /// JCS sorts keys regardless, but we also keep declaration order alphabetical
    /// as belt-and-suspenders.
    #[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Eq)]
    pub struct OuterRecordSignable {
        pub blob: String,
        pub created_at: i64,
        pub protocol_version: u16,
        pub pubkey: String,
        pub recipient: Option<String>,
        pub share_ref: String,
        pub ttl_seconds: u64,
    }

    impl From<&OuterRecord> for OuterRecordSignable {
        fn from(r: &OuterRecord) -> Self {
            OuterRecordSignable {
                blob: r.blob.clone(),
                created_at: r.created_at,
                protocol_version: r.protocol_version,
                pubkey: r.pubkey.clone(),
                recipient: r.recipient.clone(),
                share_ref: r.share_ref.clone(),
                ttl_seconds: r.ttl_seconds,
            }
        }
    }

    /// Compute the 128-bit share_ref, hex-encoded to 32 chars (D-06, PAYL-05).
    pub fn share_ref_from_bytes(ciphertext: &[u8], created_at: i64) -> String {
        let mut hasher = Sha256::new();
        hasher.update(ciphertext);
        hasher.update(created_at.to_be_bytes());
        let digest = hasher.finalize();
        let mut out = String::with_capacity(SHARE_REF_HEX_LEN);
        for b in &digest[..SHARE_REF_BYTES] {
            out.push_str(&format!("{:02x}", b));
        }
        debug_assert_eq!(out.len(), SHARE_REF_HEX_LEN);
        out
    }

    /// Serialize any Serialize value to canonical JSON per RFC 8785 (JCS).
    /// Inlined here so this module does not depend on src/crypto.rs (allows Plan 03
    /// to run in parallel with Plan 02). Can be refactored to share after merge.
    fn jcs(value: &impl Serialize) -> Result<Vec<u8>, Error> {
        let mut buf = Vec::new();
        let mut ser = serde_json::Serializer::with_formatter(&mut buf, CanonicalFormatter::new());
        value.serialize(&mut ser).map_err(|e| Error::Config(format!("jcs: {}", e)))?;
        Ok(buf)
    }

    pub fn sign_record(
        signable: &OuterRecordSignable,
        keypair: &pkarr::Keypair,
    ) -> Result<String, Error> {
        let bytes = jcs(signable)?;
        // pkarr::Keypair exposes the underlying ed25519 signing key. Exact API name
        // depends on pkarr 5.0.3 — likely `.sign(&msg) -> Signature` or
        // `.secret_key()` returning an ed25519_dalek::SigningKey. Use whichever is exported.
        let sig = keypair.sign(&bytes);  // verify this is the pkarr 5.0.3 method; adapt if .secret_key().sign() is required
        Ok(base64::engine::general_purpose::STANDARD.encode(sig.to_bytes()))
    }

    pub fn verify_record(record: &OuterRecord) -> Result<(), Error> {
        // 1. Parse pubkey from z-base-32
        let pk = pkarr::PublicKey::try_from(record.pubkey.as_str())
            .map_err(|_| Error::SignatureInner)?;
        let vk = VerifyingKey::from_bytes(&pk.to_bytes())
            .map_err(|_| Error::SignatureInner)?;

        // 2. Decode signature
        let sig_bytes = base64::engine::general_purpose::STANDARD
            .decode(&record.signature)
            .map_err(|_| Error::SignatureInner)?;
        let sig = Signature::from_slice(&sig_bytes).map_err(|_| Error::SignatureInner)?;

        // 3. Build signable, JCS-serialize
        let signable = OuterRecordSignable::from(record);
        let bytes = jcs(&signable)?;

        // 4. Verify
        vk.verify_strict(&bytes, &sig).map_err(|_| Error::SignatureInner)?;

        // 5. Re-canonicalize: parse bytes back, re-serialize, assert byte-identical.
        //    Protects against the "verify accepts a mauled payload that re-serializes
        //    to something different" attack class. Pitfall #3 subcase.
        let parsed: OuterRecordSignable = serde_json::from_slice(&bytes)
            .map_err(|_| Error::SignatureCanonicalMismatch)?;
        let round = jcs(&parsed)?;
        if round != bytes {
            return Err(Error::SignatureCanonicalMismatch);
        }

        Ok(())
    }

            let mut out = String::new();

        }

        // Map pkarr errors to cipherpost error types.
        // Timeout / network → Error::Network (exit code 6)
        // NotFound → Error::NotFound (exit code 5)
        // Everything else → Error::Transport (exit code 1)

            Error::Network

            Error::NotFound
        } else {
            Error::Transport(Box::new(e))
        }

|----------|-------------|
| DHT wire → resolver → OuterRecord deserialization | Untrusted bytes from the DHT. Must be validated via PKARR outer signature (by pkarr's `resolve_most_recent`) AND the inner Ed25519 signature over JCS bytes (by `verify_record`) BEFORE any field is surfaced. |

| MockTransport in-process | Trusted for tests; the 1000-byte ceiling is enforced in the mock to prevent tests-pass-locally-fail-on-real-publish. |

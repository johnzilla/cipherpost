    **Step A — Generate the DER fixture cert.**

    Run this openssl invocation LOCALLY (commits the bytes, not the recipe) to produce a deterministic-content cert. The private key is random, so signatures differ across regenerations — the Subject/Issuer/Serial/Validity are deterministic:

    ```bash
    mkdir -p tests/fixtures
    cd tests/fixtures

    # Ed25519 self-signed cert, deterministic-content fields.
    # -not_before / -not_after flags require OpenSSL 3.2+. If on older OpenSSL,
    # substitute -set_start and -set_end or use `-days 730` with a fixed reference date.
    openssl req -x509 -newkey ed25519 -keyout /tmp/x509_fixture_key.pem \
        -out /tmp/x509_fixture_cert.pem -nodes \
        -days 730 \
        -subj "/CN=cipherpost-fixture/O=cipherpost/C=XX" \
        -set_serial 1 \
        -addext "subjectKeyIdentifier=hash" \
        2>&1

    # Convert PEM to canonical DER:
    openssl x509 -in /tmp/x509_fixture_cert.pem -outform DER -out x509_cert_fixture.der

    # Clean up the private key (we don't check it in; it's not needed).
    shred -u /tmp/x509_fixture_key.pem /tmp/x509_fixture_cert.pem 2>/dev/null || \
        rm -f /tmp/x509_fixture_key.pem /tmp/x509_fixture_cert.pem

    # Sanity check:
    openssl x509 -in x509_cert_fixture.der -inform DER -noout -subject
    # Expected output: subject=CN=cipherpost-fixture, O=cipherpost, C=XX

    openssl x509 -in x509_cert_fixture.der -inform DER -noout -fingerprint -sha256
    # Outputs the SHA-256 fingerprint — record this for the reproduction.txt.

    ls -l x509_cert_fixture.der
    # Expected size: ~180-260 bytes depending on signature bytes.
    ```

    **NOTE:** If the `-not_before` / `-not_after` flags aren't available on your openssl build, skip them and accept a 2-year validity window starting at generation time. The Phase 6 banner will show `[VALID]` if now < NotAfter, `[EXPIRED]` otherwise — for any generation in 2026, `[VALID]` is assured until 2028.

    The DER bytes are committed as-is. The `.reproduction.txt` file documents the recipe so a future reviewer can verify the format matches what the recipe would produce (modulo signature randomness).

    //! X509-01, X509-08: ingest pipeline coverage.
    //!
    //! Exercises payload::ingest::x509_cert across happy DER, happy PEM, malformed
    //! DER (BER rejection), trailing bytes, wrong PEM label, corrupt PEM body, and
    //! cross-accessor mismatch. Every Error::InvalidMaterial is checked for
    //! oracle-hygiene: Display must never contain x509-parser internal strings.

    use base64::Engine;
    use cipherpost::payload::{ingest, Material};
    use cipherpost::Error;

    const FIXTURE_DER: &[u8] = include_bytes!("fixtures/x509_cert_fixture.der");

    fn pem_armor_der(der: &[u8]) -> Vec<u8> {
        // Hand-rolled PEM armor (LF line endings).
        let b64 = base64::engine::general_purpose::STANDARD.encode(der);
        let mut out = String::new();
        out.push_str("-----BEGIN CERTIFICATE-----\n");
        for chunk in b64.as_bytes().chunks(64) {
            out.push_str(std::str::from_utf8(chunk).expect("base64 is ASCII"));
            out.push('\n');
        }
        out.push_str("-----END CERTIFICATE-----\n");
        out.into_bytes()
    }

    fn pem_armor_der_crlf(der: &[u8]) -> Vec<u8> {
        let b64 = base64::engine::general_purpose::STANDARD.encode(der);
        let mut out = String::new();
        out.push_str("-----BEGIN CERTIFICATE-----\r\n");
        for chunk in b64.as_bytes().chunks(64) {
            out.push_str(std::str::from_utf8(chunk).expect("base64 is ASCII"));
            out.push_str("\r\n");
        }
        out.push_str("-----END CERTIFICATE-----\r\n");
        out.into_bytes()
    }

    #[test]
    fn x509_cert_happy_der_produces_x509_variant() {
        let m = ingest::x509_cert(FIXTURE_DER).expect("valid DER fixture");
        match m {
            Material::X509Cert { bytes } => {
                assert_eq!(
                    bytes, FIXTURE_DER,
                    "stored bytes must equal input DER (canonical-DER invariant)"
                );
            }
            other => panic!("expected X509Cert variant, got {:?}", other),
        }
    }

    #[test]
    fn x509_cert_happy_pem_normalizes_to_der() {
        let pem = pem_armor_der(FIXTURE_DER);
        let m = ingest::x509_cert(&pem).expect("valid PEM-wrapped fixture");
        match m {
            Material::X509Cert { bytes } => {
                assert_eq!(
                    bytes, FIXTURE_DER,
                    "PEM ingest must normalize to ORIGINAL DER (not the PEM bytes)"
                );
            }
            other => panic!("expected X509Cert variant, got {:?}", other),
        }
    }

    #[test]

    /// `Material::SshKey { bytes: <canonical re-encoded> }`.
    ///
    /// Pipeline:
    ///   1. Sniff: after `trim_start()` over ASCII whitespace, if the input does
    ///      NOT start with `-----BEGIN OPENSSH PRIVATE KEY-----`, reject with
    ///      `Error::SshKeyFormatNotSupported` (D-P7-12). This catches legacy PEM
    ///      (`-----BEGIN RSA/DSA/EC PRIVATE KEY-----`), RFC 4716 (`---- BEGIN
    ///      SSH2 ENCRYPTED PRIVATE KEY ----`), FIDO (`-----BEGIN OPENSSH-FIDO
    ///      PRIVATE KEY-----`), and arbitrary garbage in one rejection.
    ///   2. Parse: `ssh-key::PrivateKey::from_openssh(input_str)` validates the
    ///      OpenSSH v1 framing + algorithm + key-data structure. Parse failure
    ///      (header IS OpenSSH-v1 but body is malformed) → return
    ///      `Error::InvalidMaterial { variant: "ssh_key", reason: "malformed
    ///      OpenSSH v1 blob" }`. This is the only error path that uses
    ///      `InvalidMaterial` for SSH — format-rejections use the dedicated variant.
    ///   3. Re-encode: `parsed.to_openssh(LineEnding::LF)` produces the canonical
    ///      OpenSSH v1 output as a `Zeroizing<String>`. We store the UTF-8 bytes
    ///      of this string in `Material::SshKey.bytes` (D-P7-11 canonical wire blob).
    ///   4. Return `Material::SshKey { bytes }`.
    ///
    /// The `Error::InvalidMaterial { reason: "trailing bytes after OpenSSH v1 blob" }`
    /// path (WR-01 mirror) is REACHED IF ssh-key's `from_openssh` accepts trailing
    /// content after the `-----END OPENSSH PRIVATE KEY-----` line. Empirically check
    /// during implementation — if ssh-key rejects natively, we do NOT need an
    /// explicit trailing-bytes assertion (the canonical re-encode skips trailers
    /// for free). If ssh-key accepts, add a re-parse-then-compare-length check.
    pub fn ssh_key(raw: &[u8]) -> Result<Material, Error> {
        // --- Step 1: format sniff. ---------------------------------------
        let first_non_ws = raw.iter().position(|b| !b.is_ascii_whitespace()).unwrap_or(raw.len());
        let trimmed = &raw[first_non_ws..];
        if !trimmed.starts_with(b"-----BEGIN OPENSSH PRIVATE KEY-----") {
            return Err(Error::SshKeyFormatNotSupported);
        }

        // --- Step 2: parse via ssh-key. ----------------------------------
        // ssh-key::PrivateKey::from_openssh accepts &str. raw is &[u8] of UTF-8
        // (PEM is ASCII; safe). Convert via std::str::from_utf8.
        let raw_str = std::str::from_utf8(raw).map_err(|_| Error::InvalidMaterial {
            variant: "ssh_key".into(),
            reason: "malformed OpenSSH v1 blob".into(),
        })?;
        let parsed = ssh_key::PrivateKey::from_openssh(raw_str).map_err(|_| {
            Error::InvalidMaterial {
                variant: "ssh_key".into(),
                reason: "malformed OpenSSH v1 blob".into(),
            }
        })?;

        // --- Step 3: re-encode canonically. ------------------------------
        // ssh-key 0.6.7 returns Zeroizing<String>. Convert to bytes and store.
        // LineEnding::LF is the OpenSSH v1 standard.
        let encoded = parsed
            .to_openssh(ssh_key::LineEnding::LF)
            .map_err(|_| Error::InvalidMaterial {
                variant: "ssh_key".into(),
                reason: "malformed OpenSSH v1 blob".into(),
            })?;

        // --- Step 4: trailing-bytes check (WR-01 mirror). ----------------
        // If ssh-key's from_openssh consumed a strict prefix of `raw_str` and
        // ignored content after `-----END OPENSSH PRIVATE KEY-----`, an attacker
        // could append junk to make a second share_ref for the same key. The
        // canonical re-encode discards trailers — so if `encoded.as_bytes()`
        // matches `raw` modulo line-ending normalization, we're fine. If they
        // differ in a way that suggests trailer truncation, reject.
        //
        // Simplest correct check: re-parse `encoded` and assert it round-trips
        // (already covered by the byte-determinism test below). For ingest
        // robustness, also verify that any input bytes BEYOND the END marker
        // are all-whitespace.
        let end_marker = b"-----END OPENSSH PRIVATE KEY-----";
        if let Some(end_pos) = raw.windows(end_marker.len()).position(|w| w == end_marker) {
            let after_end = &raw[end_pos + end_marker.len()..];
            if !after_end.iter().all(|b| b.is_ascii_whitespace()) {
                return Err(Error::InvalidMaterial {
                    variant: "ssh_key".into(),
                    reason: "trailing bytes after OpenSSH v1 blob".into(),
                });
            }
        }
        // (If end_marker not found, ssh-key's from_openssh would have rejected
        // already in Step 2, so we can't reach here.)

        Ok(Material::SshKey { bytes: encoded.as_bytes().to_vec() })
    }

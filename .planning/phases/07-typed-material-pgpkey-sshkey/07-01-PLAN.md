    ```rust
    /// No canonical re-encode — the binary packet stream IS canonical (RFC 4880 §4.2 +
    /// RFC 9580 §4.2). Re-encoding through rpgp could alter insignificant bits that
    /// would drift `share_ref` across sender toolchains; storing input bytes verbatim
    /// (after validation) is the correctness-preserving choice.
    ///
    /// Pipeline:
    ///   1. Sniff: after `trim_start()` over ASCII whitespace, reject if the input
    ///      begins with `-----BEGIN PGP` (catches PUBLIC KEY BLOCK and PRIVATE KEY
    ///      BLOCK both — D-P7-05).
    ///   2. Parse: iterate top-level packets via rpgp's low-level PacketParser.
    ///      Count tag-5 (Secret-Key) + tag-6 (Public-Key) at the TOP LEVEL — do NOT
    ///      recurse into subpacket data. Tag-7 (Secret-Subkey) and tag-14 (Public-
    ///      Subkey) are NOT counted (subkeys are legitimate; we're rejecting
    ///      keyrings, not subkey-rich single keys).
    ///   3. If primary count is 0: malformed (no primary key packet). Reject.
    ///   4. If primary count > 1: keyring. Reject per D-P7-06 / PGP-03.
    ///   5. Assert the parser consumed the entire input — trailing bytes after the
    ///      last valid packet are rejected (WR-01 invariant mirror).
    ///   6. Return `Material::PgpKey { bytes: raw.to_vec() }`.
    ///
    /// Error contract: every failure path returns `Error::InvalidMaterial { variant:
    /// "pgp_key", reason: "<short generic>" }`. NEVER wrap a `pgp` crate error — the
    /// `reason: String` is the oracle-hygiene gate (PGP-08 / D-P7-09 extension).
    pub fn pgp_key(raw: &[u8]) -> Result<Material, Error> {
        // --- Step 1: armor-prefix sniff. ---------------------------------
        let first_non_ws = raw.iter().position(|b| !b.is_ascii_whitespace()).unwrap_or(raw.len());
        let trimmed = &raw[first_non_ws..];
        if trimmed.starts_with(b"-----BEGIN PGP") {
            return Err(Error::InvalidMaterial {
                variant: "pgp_key".into(),
                reason: "ASCII-armored input rejected — supply binary packet stream".into(),
            });
        }

        // --- Step 2: parse top-level packets, count primaries. -----------
        // Planner guidance: rpgp 0.19.0 exposes packet-level iteration via
        // `pgp::packet::PacketParser::new(std::io::Cursor::new(raw))` yielding
        // `Result<Packet, Error>`. Each top-level packet has a `.tag()` returning
        // `pgp::types::Tag` with variants including `PublicKey` (= RFC tag 6),
        // `SecretKey` (= RFC tag 5), `PublicSubkey` (= tag 14), `SecretSubkey`
        // (= tag 7). Count PublicKey + SecretKey. If the low-level API differs,
        // fall back to `pgp::composed::Message::from_bytes` + `from_bytes_many`
        // and check `keys.len() > 1` OR use `pgp::composed::StandaloneSignature` +
        // `pgp::composed::SignedPublicKey::from_bytes_many` to detect multi-primary.
        //
        // The EXACT API is to be determined at implementation time; the acceptance
        // criterion is the six behavior tests above, not the internal API choice.

        use std::io::Cursor;
        let cursor = Cursor::new(raw);
        let mut primary_count = 0usize;
        let mut total_packets = 0usize;
        let mut bytes_consumed = 0usize;

        // REPLACE this block with the actual pgp 0.19.0 packet-iteration API.
        // Pseudo-code template (the executor adapts to the real API):
        let parser = pgp::packet::PacketParser::new(cursor);
        for packet_result in parser {
            let packet = packet_result.map_err(|_| Error::InvalidMaterial {
                variant: "pgp_key".into(),
                reason: "malformed PGP packet stream".into(),
            })?;
            total_packets += 1;
            match packet.tag() {
                pgp::types::Tag::PublicKey | pgp::types::Tag::SecretKey => {
                    primary_count += 1;
                }
                _ => {}
            }
            // Track bytes consumed for the trailing-bytes check (rpgp may not expose
            // a cursor position directly; if not, fall back to counting via
            // packet.header_len() + packet.body_len() or similar per-packet API).
            bytes_consumed += packet.packet_length().map(|l| l as usize).unwrap_or(0);
        }

        if total_packets == 0 {
            return Err(Error::InvalidMaterial {
                variant: "pgp_key".into(),
                reason: "malformed PGP packet stream".into(),
            });
        }
        if primary_count == 0 {
            return Err(Error::InvalidMaterial {
                variant: "pgp_key".into(),
                reason: "malformed PGP packet stream".into(),
            });
        }
        if primary_count > 1 {
            return Err(Error::InvalidMaterial {
                variant: "pgp_key".into(),
                reason: format!(
                    "PgpKey must contain exactly one primary key; keyrings are not supported in v1.1 (found {} primary keys)",
                    primary_count
                ),
            });
        }

        // --- Step 3: trailing-bytes check (WR-01 invariant). ------------
        // If rpgp's PacketParser doesn't expose bytes-consumed, use the alternative:
        // re-run the parse collecting (start, end) offsets and assert the last packet
        // end == raw.len(). If neither is feasible, use a smaller "no bytes left
        // after final packet" heuristic via a second pass.
        if bytes_consumed != raw.len() {
            return Err(Error::InvalidMaterial {
                variant: "pgp_key".into(),
                reason: "trailing bytes after PGP packet stream".into(),
            });
        }

        Ok(Material::PgpKey { bytes: raw.to_vec() })
    }
    ```

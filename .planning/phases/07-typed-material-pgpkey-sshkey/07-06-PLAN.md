    ```rust
    /// Render an SSH acceptance-banner subblock from the canonical OpenSSH v1
    /// wire blob (`Material::SshKey.bytes` after Plan 05's re-encode).
    ///
    /// Returns a multi-line String (no leading or trailing newline). Field
    /// ordering: Key (algorithm + size + optional `[DEPRECATED]`) → Fingerprint
    /// (SHA-256 base64-unpadded) → Comment ([sender-attested] + truncated).
    ///
    /// NO SECRET-key warning line: OpenSSH v1 ALWAYS contains a private key
    /// (the format is for private keys); warning every time would be noise.
    /// D-P7-14 chose the lighter `[DEPRECATED]` treatment for legacy algorithms
    /// instead.
    ///
    /// Caller (run_receive in Plan 07) passes the returned string through
    /// `Option<&str>` to TtyPrompter's preview_subblock parameter.
    pub fn render_ssh_preview(bytes: &[u8]) -> Result<String, Error> {
        use std::fmt::Write as _;

        // --- Step 1: parse via ssh-key. ---------------------------------
        // bytes are UTF-8 PEM-armored OpenSSH v1 (per Plan 05's canonical re-encode).
        let raw_str = std::str::from_utf8(bytes).map_err(|_| Error::InvalidMaterial {
            variant: "ssh_key".into(),
            reason: "malformed OpenSSH v1 blob".into(),
        })?;
        let key = ssh_key::PrivateKey::from_openssh(raw_str).map_err(|_| {
            Error::InvalidMaterial {
                variant: "ssh_key".into(),
                reason: "malformed OpenSSH v1 blob".into(),
            }
        })?;

        // --- Step 2: extract fields. ------------------------------------
        let public_key = key.public_key();
        let algorithm_str = public_key.algorithm().as_str(); // "ssh-ed25519" | "ssh-rsa" | "ssh-dss" | "ecdsa-sha2-nistp256" | ...
        let bits: Option<u32> = ssh_public_key_bit_size(public_key);
        let deprecated = is_deprecated_ssh_algorithm(algorithm_str, bits);

        // SHA-256 fingerprint — ssh-key 0.6.7 returns Fingerprint enum whose
        // Display impl produces "SHA256:<b64-unpadded>" matching ssh-keygen -lf.
        let fingerprint = public_key.fingerprint(ssh_key::HashAlg::Sha256);
        let fingerprint_str = format!("{}", fingerprint);

        let comment_raw = key.comment();
        let comment_display = if comment_raw.is_empty() {
            "(none)".to_string()
        } else {
            truncate_display(comment_raw, SSH_COMMENT_TRUNC_LIMIT)
        };

        // --- Step 3: format. --------------------------------------------
        let separator = format!("--- SSH {}", "-".repeat(SSH_SEPARATOR_DASH_COUNT));
        let key_line = match (bits, deprecated) {
            (Some(b), true) => format!("{} {} [DEPRECATED]", algorithm_str, b),
            (Some(b), false) => format!("{} {}", algorithm_str, b),
            (None, true) => format!("{} [DEPRECATED]", algorithm_str),
            (None, false) => algorithm_str.to_string(),
        };

        let mut out = String::new();
        out.push_str(&separator);
        out.push('\n');
        writeln!(out, "Key:         {}", key_line).expect("String write");
        writeln!(out, "Fingerprint: {}", fingerprint_str).expect("String write");
        // Last line — no trailing newline.
        write!(out, "Comment:     [sender-attested] {}", comment_display).expect("String write");
        Ok(out)
    }

    /// Extract the bit size of an SSH public key. ssh-key 0.6.7's Algorithm
    /// enum exposes the type but not the bits directly for RSA — we compute
    /// from the modulus length, OR rely on the embedded `KeyData::bits()` if
    /// available.
    ///
    /// EXECUTOR: substitute with the real ssh-key 0.6.7 API. Candidates:
    ///   - public_key.key_data() returns &KeyData; KeyData::bits() may be
    ///     available as `pub fn bits(&self) -> u32`
    ///   - For Ed25519: 256
    ///   - For RSA: extract from modulus byte length × 8
    ///   - For ECDSA P-256/P-384/P-521: 256/384/521
    ///   - For DSA: traditionally 1024 (legacy), but ssh-key may not expose
    fn ssh_public_key_bit_size(public_key: &ssh_key::PublicKey) -> Option<u32> {
        // Pseudo-impl — adapt to real API:
        //   public_key.key_data().bits()  // if available
        //   OR match on public_key.algorithm() and dispatch per-variant
        //
        // For Ed25519: Some(256)
        // For RSA: derive from public_key.key_data() if it's the RSA variant
        //   (typically rsa.n.as_bytes().len() * 8 — modulus bit length)
        // For ECDSA: pattern-match on the curve
        // For DSA: 1024 (legacy default; SSH-DSS keys are 1024 bits)
        // Unknown: None
        None  // EXECUTOR placeholder — real implementation determines bits per algorithm
    }
    ```
